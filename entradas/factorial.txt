int factorial(int n)
{   
    int resultF;
    output(n);
    if (n <= 2){
        return 1;
    } 
    else {
      resultF = factorial(n - 1);
      output(resultF);
      return (n * resultF);  
    } 
}

void main(void)
{
    int number;
    int result;
    
    number = input();
    result = factorial(number);
    output(result);
}
